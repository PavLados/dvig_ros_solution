// обработчик прерывания энкодера - правый мотор
void encoder2() {

  if (digitalRead(ENCODER_2_B) == HIGH){ // если на контакте ENCODER_B уровень high увеличиваем count
    encoder_count2++; }// increment the count
  else {// иначе уменьшаем count
    encoder_count2--;}  // decrement the count  

}// обработчик прерывания энкодера - левый мотор
void encoder1() { 
  
  if (digitalRead(ENCODER_1_B) == HIGH) // если на контакте ENCODER_B уровень high увеличиваем count
    encoder_count1--; // increment the count
  else // иначе уменьшаем count
    encoder_count1++;  // decrement the count   
}
//функция обратной связи ошибки скорости моторов
float computeP(float setpoint, float input, float dt) {
    float error = setpoint + (setpoint - input);

    //if (abs(input)=0){return setpoint;}
    if ((error > setpoint)and(input==0)) {return 100;}
    if (error>255){return 255;}
    if (error<-255){return -255;}
    return error * Kp;
}
